10  import org.apache.commons.exec.CommandLine;
11  import org.apache.commons.exec.DefaultExecutor;

13  import org.apache.commons.exec.PumpStreamHandler;
14  import org.apache.maven.plugin.AbstractMojo;
15  import org.apache.maven.plugin.MojoExecutionException;

17  public abstract class AbstractDebianMojo extends AbstractMojo
18  {
19  	private static final String SKIP_DEB_PROPERTY = "skipDeb";
20  	private static final String RUN_DEB_PROPERTY = "runDeb";
21  
22  	/**
23  	 * @parameter expression="${deb.package.name}" default-value="${project.artifactId}"
24  	 */
25  	protected String packageName;
26  
27  	/**
28  	 * @parameter expression="${deb.package.version}" default-value="${project.version}"
29  	 */
30  	private String packageVersion;
31  
32  	/**
33  	 * @parameter expression="${deb.package.revision}" default-value="1"
34  	 */
35  	protected String packageRevision;
36  
37  	/**
38  	 * @parameter expression="${deb.package.architecture}" default-value="all"
39  	 */
40  	protected String packageArchitecture;
41  
42  	/**
43  	 * @parameter expression="${deb.maintainer.name}" default-value="${project.developers[0].name}"
44  	 */
45  	protected String maintainerName;
46  
47  	/**
48  	 * @parameter expression="${deb.maintainer.email}" default-value="${project.developers[0].email}"
49  	 */
50  	protected String maintainerEmail;
51  
52  	/** @parameter default-value="${basedir}/src/deb" */
53  	protected File sourceDir;
54  
55  	/** @parameter default-value="${basedir}/target" */
56  	protected File targetDir;
57  
58  	/** @parameter default-value="${basedir}/target/deb" */
59  	protected File stageDir;
60  
61  	/**
62  	 * @parameter expression="${deb.package.snapshotRevFile}"
63  	 * @since 1.0.5
64  	 */
65  	private File snapshotRevisionFile = null;
66  
67  	private static final DateFormat datefmt = new SimpleDateFormat("yyyyMMddHHmm");
68  
69  	private String snapshotRevision = null;
70  
71  	protected String processVersion(String version)
72  	{
73  		if (snapshotRevision == null)
74  		{
75  			Date revtime = snapshotRevisionFile != null
76  					? new Date(snapshotRevisionFile.lastModified())
77  					: new Date();
78  
79  			snapshotRevision = "+" + datefmt.format(revtime);
80  		}
81  
82  		return version.replaceAll("-SNAPSHOT", snapshotRevision);
83  	}
84  
85  	protected String[] processVersion(String[] versions)
86  	{
87  		String[] result = new String[versions.length];
88  		for (int i=0 ; i<versions.length ; i++)
89  			result[i] = processVersion(versions[i]);
90  
91  		return result;
92  	}
93  
94  	protected String getPackageVersion()
95  	{
96  		return processVersion(packageVersion);
97  	}
98  
99  	protected File getPackageFile()
100 	{
101 		return new File(targetDir, String.format("%s_%s-%s_all.deb", packageName, getPackageVersion(), packageRevision));
102 	}
103 
104 	protected void runProcess(String[] cmd, boolean throw_on_failure) throws ExecuteException, IOException, MojoExecutionException
105 	{
106 		CommandLine cmdline = new CommandLine(cmd[0]);
107 		cmdline.addArguments(Arrays.copyOfRange(cmd, 1, cmd.length));
108 
109 		getLog().info("Start process: "+cmdline);
110 
111 		PumpStreamHandler streamHandler = new PumpStreamHandler(new LogOutputStream(getLog()));
112 		DefaultExecutor exec = new DefaultExecutor();
113 		exec.setStreamHandler(streamHandler);
114 		int exitval = exec.execute(cmdline);
115 		if (exitval != 0)
116 		{
117 			getLog().warn("Exit code "+exitval);
118 			
119 			if (throw_on_failure)
120 				throw new MojoExecutionException("Process returned non-zero exit code: "+cmdline);
121 		}
122 	}
123 
124 	protected abstract void executeDebMojo() throws MojoExecutionException;
125 
126 	public final void execute() throws MojoExecutionException
127 	{
128 		if (System.getProperties().containsKey(RUN_DEB_PROPERTY))
129 		{
130 			getLog().info("debian-maven execution forced (-DrunDeb)");
131 			executeDebMojo();
132 		}
133 		else if (System.getProperties().containsKey(SKIP_DEB_PROPERTY))
134 			getLog().info("debian-maven execution skipped (-DskipDeb)");
135 		else if (!System.getProperty("os.name").equals("Linux"))
136 			getLog().warn("debian-maven execution skipped (non-linux OS)");
137 		else
138 			executeDebMojo();
139 	}
140 }
