155 	private void writeIncludeFile(File targetLibDir, String artifactId, String version, Collection<String> dependencies) throws IOException, MojoExecutionException
156 	{
157 		if (dependencies == null)
158 			dependencies = Collections.emptySet();
159 
160 		File deplist = new File(targetLibDir, String.format("%s-%s.inc", artifactId, version));
161 		FileWriter out = new FileWriter(deplist);
162 		try
163 		{
164 			out.write(String.format("artifacts=%s\n", StringUtils.join(new HashSet<String>(dependencies), ":")));
165 		}
166 		finally
167 		{
168 			out.close();
169 		}
170 
171 		createSymlink(new File(targetLibDir, String.format("%s.inc", artifactId)), deplist.getName());
172 	}

174 	private boolean includeArtifact(Artifact a)
175 	{
176 		boolean doExclude = excludeArtifacts != null && (a.getDependencyTrail() == null || Collections.disjoint(a.getDependencyTrail(), excludeArtifacts));
177 		if (doExclude)
178 			return false;
179 
180 		if (includeArtifacts == null)
181 			return true;
182 
183 		if (a.getDependencyTrail() == null)
184 			return true;
185 
186 		return Collections.disjoint(a.getDependencyTrail(), includeArtifacts);
187 	}

189 	private File copyArtifact(Artifact a, File targetLibDir) throws IOException, MojoExecutionException
190 	{
191 		if (a.getFile() == null)
192 			throw new MojoExecutionException(String.format("No file was built for required artifact: %s:%s:%s", a.getGroupId(), a.getArtifactId(), a.getVersion()));
193 
194 		getLog().info(String.format("Artifact: %s", a.getFile().getPath()));
195 		File src = a.getFile();
196 		File trg = new File(targetLibDir, src.getName());
197 		FileUtils.copyFile(src, trg);
198 
199 		//TODO: which version should we use? trying both versions for now...
200 		String linkname = src.getName().replaceFirst("-"+a.getBaseVersion(), "");
201 		if (linkname.equals(src.getName()))
202 			linkname = linkname.replaceFirst("-"+a.getVersion(), "");
203 
204 		if (!linkname.equals(src.getName()))
205 			createSymlink(new File(targetLibDir, linkname), a.getFile().getName());
206 
207 		return trg;
208 	}

210 	@SuppressWarnings("unchecked")
211 	private void copyAttachedArtifacts() throws FileNotFoundException, IOException, MojoExecutionException
212 	{
213 		if (!includeAttachedArtifacts)
214 		{
215 			getLog().info("Skipping attached project artifacts.");
216 			return;
217 		}
218 
219 		getLog().info("Copying attached project artifacts.");
220 		File targetLibDir = createTargetLibDir();
221 
222 		for (Artifact a : (Collection<Artifact>)project.getAttachedArtifacts())
223 			copyArtifact(a, targetLibDir);
224 	}

226 	@SuppressWarnings("unchecked")
227 	private void copyArtifacts() throws FileNotFoundException, IOException, MojoExecutionException
228 	{
229 		if (excludeAllArtifacts)
230 		{
231 			getLog().info("Skipping regular project artifacts and dependencies.");
232 			return;
233 		}
234 
235 		File targetLibDir = createTargetLibDir();
236 
237 		Collection<Artifact> artifacts = new ArrayList<Artifact>();
238 
239 		// consider the current artifact only if it exists (e.g. pom, war packaging generates no artifact)
240 		if (project.getArtifact().getFile() != null)
241 			artifacts.add(project.getArtifact());
242 
243 		if (excludeAllDependencies)
244 			getLog().info("Copying regular project artifacts but not dependencies.");
245 		else
246 		{
247 			getLog().info("Copying regular project artifacts and dependencies.");
248 			for (Artifact a : (Collection<Artifact>)project.getArtifacts())
249 			{
250 				if (a.getScope().equals("runtime") || a.getScope().equals("compile"))
251 					artifacts.add(a);
252 			}
253 		}
254 
255 		/*
256 		 * TODO: this code doesn't work as it should due to limitations of Maven API; see also:
257 		 * http://jira.codehaus.org/browse/MNG-4831
258 		 */
259 
260 		Map<String,Artifact> ids = new HashMap<String,Artifact>();
261 		for (Artifact a : artifacts)
262 			ids.put(a.getId(), a);
263 
264 		MultiMap<Artifact,String> deps = new MultiHashMap<Artifact,String>();
265 		for (Artifact a : artifacts)
266 		{
267 			if (includeArtifact(a))
268 			{
269 				File trg = copyArtifact(a, targetLibDir);
270 
271 				if (a.getDependencyTrail() != null)
272 				{
273 					for (String id : a.getDependencyTrail())
274 					{
275 						Artifact depending = ids.get(id);
276 						if (depending != null)
277 							deps.put(depending, trg.getPath().substring(stageDir.getPath().length()));
278 					}
279 				}
280 			}
281 		}
282 
283 		for (Artifact a : artifacts)
284 		{
285 			if (includeArtifact(a))
286 				writeIncludeFile(targetLibDir, a.getArtifactId(), a.getVersion(), deps.get(a));
287 		}
288 	}

290 	private void generateCopyright() throws IOException
291 	{
292 		File targetDocDir = new File(stageDir, "usr/share/doc/" + packageName);

294 
295 		PrintWriter out = new PrintWriter(new FileWriter(new File(targetDocDir, "copyright")));

298 		out.println();
299 		out.printf("Copyright %d %s\n", Calendar.getInstance().get(Calendar.YEAR), projectOrganization);

308 	private void generateControl(File target) throws IOException
309 	{
310 		getLog().info("Generating control file: "+target);
311 		PrintWriter out = new PrintWriter(new FileWriter(target));
312 
313 		out.println("Package: "+packageName);
314 		out.println("Version: "+getPackageVersion());
315 
316 		if (packageSection != null)
317 			out.println("Section: "+packageSection);
318 		if (packagePriority != null)
319 			out.println("Priority: "+packagePriority);
320 		out.println("Architecture: "+packageArchitecture);
321 		if (packageDependencies != null && packageDependencies.length > 0)
322 			out.println("Depends: " + StringUtils.join(processVersion(packageDependencies), ", "));
323 
324 		out.printf("Installed-Size: %d\n", 1 + FileUtils.sizeOfDirectory(stageDir) / 1024);
325 
326 		if (maintainerName != null || maintainerEmail != null)
327 		{
328 			out.print("Maintainer:");
329 			if (maintainerName != null)
330 				out.print(" "+maintainerName);
331 			if (maintainerEmail != null)
332 				out.printf(" <%s>", maintainerEmail);
333 			out.println();
334 		}
335 
336 		if (projectUrl != null)
337 			out.println("Homepage: "+projectUrl);
338 
339 		if (packageTitle != null) {
340 			if (packageTitle.length() > 60)
341 			{
342 				getLog().warn("Package title will be truncated to the upper limit of 60 characters.");
343 				out.println("Description: "+packageTitle.substring(0, 60));
344 			}
345 			else
346 				out.println("Description: "+packageTitle);
347 
348 			out.println(getFormattedDescription());
349 		}
350 
351 		out.close();
352 	}

362 	private void generateConffiles(File target) throws IOException
363 	{
364 		List<String> conffiles = new Vector<String>();
365 
366 		File configDir = new File(stageDir, "etc");
367 		if (configDir.exists())
368 		{
369 			Collection<File> files = FileUtils.listFiles(configDir, null, true);
370 			for (File f : files)
371 			{
372 				if (f.isFile())
373 					conffiles.add(f.toString().substring(stageDir.toString().length()));
374 			}
375 		}
376 
377 		if (conffiles.size() > 0)
378 		{
379 			PrintWriter out = new PrintWriter(new FileWriter(target));
380 			for (String fname : conffiles)
381 				out.println(fname);
382 			out.close();
383 		}
384 	}

386 	private void generateMd5Sums(File target) throws IOException
387 	{
388 		PrintWriter out = new PrintWriter(new FileWriter(target));
389 		
390 		Collection<File> files = FileUtils.listFiles(stageDir, null, true);
391 		for (File f : files)
392 		{
393 			// check whether the file is a non-regular file
394 			if (!f.isFile())
395 				continue;
396 			
397 			// check whether the file is a possible link
398 			if (!f.getAbsolutePath().equals(f.getCanonicalPath()))
399 				continue;
400 
401 			String fname = f.toString().substring(stageDir.toString().length() + 1);
402 			if (!fname.startsWith("DEBIAN"))
403 			{
404 				FileInputStream fis = new FileInputStream(f);
405 				String md5 = DigestUtils.md5Hex(fis);
406 				fis.close();
407 				
408 				out.printf("%s  %s\n", md5, fname);
409 			}
410 		}
411 		
412 		out.close();
413 	}

415 	private void generateManPages() throws MojoExecutionException, ExecuteException, IOException
416 	{
417 		File source = new File(sourceDir, "man");
418 		if (!source.exists())
419 		{
420 			getLog().info("No manual page directory found: "+source);
421 			return;
422 		}
423 
424 		int npages = 0;
425 		Collection<File> files = FileUtils.listFiles(source, null, true);
426 		for (File f : files)
427 		{
428 			if (f.isFile() && f.getName().matches(".*[.][1-9]$"))
429 			{
430 				char section = f.getName().charAt(f.getName().length()-1);
431 				File target = new File(stageDir, String.format("usr/share/man/man%c/%s.gz", section, f.getName()));
432 				target.getParentFile().mkdirs();
433 
434 				CommandLine cmdline = new CommandLine("groff");
435 				cmdline.addArguments(new String[]{"-man", "-Tascii", f.getPath()});
436 
437 				getLog().info("Start process: "+cmdline);
438 
439 				GZIPOutputStream os = new GZIPOutputStream(new FileOutputStream(target));
440 				try
441 				{
442 					PumpStreamHandler streamHandler = new PumpStreamHandler(os, new LogOutputStream(getLog()));
443 					DefaultExecutor exec = new DefaultExecutor();
444 					exec.setWorkingDirectory(f.getParentFile());
445 					exec.setStreamHandler(streamHandler);
446 					int exitval = exec.execute(cmdline);
447 					if (exitval == 0)
448 						getLog().info("Manual page generated: "+target.getPath());
449 					else
450 					{
451 						getLog().warn("Exit code "+exitval);
452 						throw new MojoExecutionException("Process returned non-zero exit code: "+cmdline);
453 					}
454 				}
455 				finally
456 				{
457 					os.close();
458 				}
459 
460 				npages++;
461 			}
462 		}
463 
464 		if (npages == 0)
465 			getLog().info("No manual pages found in directory: "+source);
466 	}

479 	protected void executeDebMojo() throws MojoExecutionException
480 	{
481 		checkDeprecated(includeJar != null, "includeJar");
482 		checkDeprecated(includeJars != null && includeJars.length > 0, "includeJars");
483 		checkDeprecated(excludeAllJars != null, "excludeAllJars");
484 
485 		File targetDebDir = new File(stageDir, "DEBIAN");
486 		if (!targetDebDir.exists() && !targetDebDir.mkdirs())
487 			throw new MojoExecutionException("Unable to create directory: "+targetDebDir);
488 
489 		try
490 		{
491 			generateManPages();
492 			copyAttachedArtifacts();
493 			copyArtifacts();
494 			generateCopyright();
495 			generateConffiles(new File(targetDebDir, "conffiles"));
496 			generateControl(new File(targetDebDir, "control"));
497 			generateMd5Sums(new File(targetDebDir, "md5sums"));
498 			generatePackage();
499 		}
500 		catch (IOException e)
501 		{
502 			getLog().error(e.toString());
503 			throw new MojoExecutionException(e.toString());
504 		}
