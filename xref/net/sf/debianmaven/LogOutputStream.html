8   public class LogOutputStream extends OutputStream
9   {
10  	private final Log backend;
11  	private StringBuffer buffer = new StringBuffer();
12  	
13  	public LogOutputStream(Log backend)
14  	{
15  		this.backend = backend;
16  	}
17  
18  	@Override
19  	public void write(int ch) throws IOException
20  	{
21  		if ((char)ch == '\n')
22  		{
23  			backend.warn(buffer.toString());
24  			buffer.setLength(0);
25  		}
26  		else
27  			buffer.append((char)ch);
28  	}
29  }
