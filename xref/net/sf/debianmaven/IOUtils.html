42  	/**
43  	 * Applies the patterns to fix file permissions.
44  	 * @param log for logging output
45  	 * @param dir the directory to process
46  	 * @param fixPermissions the permissions to apply
47  	 * @throws IOException if application of permissions fails
48  	 */
49  	public static void applyFixPermissions(Log log, File dir, List<FixPermission> fixPermissions) throws IOException
50  	{
51  		log.debug("Fixing permissions in: " + dir);
52  
53  		// directory itself
54  		for (FixPermission fixPermission: fixPermissions)
55  		{
56  			if (fixPermission.appliesTo(dir))
57  			{
58  				log.debug("'" + fixPermission + "' applies to: " + dir);
59  				fixPermission.applyTo(dir);
60  			}
61  		}
62  
63  		// iterate files/dirs
64  		File[] files = dir.listFiles();
65  		if (files != null)
66  		{
67  			for (File file: files)
68  			{
69  				if (file.isDirectory())
70  					applyFixPermissions(log, file, fixPermissions);
71  				else
72  				{
73  					for (FixPermission fixPermission: fixPermissions)
74  					{
75  						if (fixPermission.appliesTo(file))
76  						{
77  							log.debug("'" + fixPermission + "' applies to: " + file);
78  							fixPermission.applyTo(file);
79  						}
80  					}
81  				}
82  			}
83  		}
84  	}

86  	/**
87  	 * Copies or moves files and directories (recursively).
88  	 * If targetLocation does not exist, it will be created.
89  	 * <br><br>
90  	 * Original code from <a href="http://www.java-tips.org/java-se-tips/java.io/how-to-copy-a-directory-from-one-location-to-another-loc.html" target="_blank">Java-Tips.org</a>.
91  	 *
92  	 * @param log for logging
93  	 * @param sourceLocation	the source file/dir
94  	 * @param targetLocation	the target file/dir
95  	 * @param move		if true then the source files/dirs get deleted
96  	 * 				as soon as copying finished
97  	 * @param atomic		whether to perform an atomic move operation
98  	 * @param include   the regular expression pattern for including files/dirs
99  	 * @return			false if failed to delete when moving or failed to create target directory
100 	 * @throws IOException	if copying/moving fails
101 	 */
102 	public static boolean copyOrMove(Log log, File sourceLocation, File targetLocation, boolean move, boolean atomic, Pattern include) throws IOException
103 	{
104 		String[] children;
105 		int i;
106 		Path source;
107 		Path target;
108 
109 		if (sourceLocation.isDirectory()) {
110 			log.debug("Entering source dir: " + sourceLocation);
111 			if (!targetLocation.exists()) {
112 				if (!targetLocation.mkdirs() && !targetLocation.exists())
113 				{
114 					log.error("Failed to create target directory: " + targetLocation);
115 					return false;
116 				}
117 			}
118 
119 			children = sourceLocation.list();
120 			for (i = 0; i < children.length; i++) {
121 				if (include.matcher(new File(sourceLocation.getAbsoluteFile(), children[i]).getAbsolutePath()).matches())
122 				{
123 					if (!copyOrMove(
124 						log,
125 						new File(sourceLocation.getAbsoluteFile(), children[i]),
126 						new File(targetLocation.getAbsoluteFile(), children[i]),
127 						move, atomic, include))
128 						return false;
129 				}
130 			}
131 
132 			if (move)
133 				return sourceLocation.delete();
134 			else
135 				return true;
136 		}
137 		else {
138 			source = FileSystems.getDefault().getPath(sourceLocation.getAbsolutePath());
139 			if (include.matcher(source.toAbsolutePath().toString()).matches())
140 			{
141 				if (targetLocation.isDirectory())
142 					target = FileSystems.getDefault().getPath(targetLocation.getAbsolutePath() + File.separator + sourceLocation.getName());
143 				else
144 					target = FileSystems.getDefault().getPath(targetLocation.getAbsolutePath());
145 				if (move)
146 				{
147 					log.debug("Moving '" + sourceLocation + "' to '" + target + "'");
148 					if (atomic)
149 						Files.move(source, target, StandardCopyOption.REPLACE_EXISTING, StandardCopyOption.ATOMIC_MOVE);
150 					else
151 						Files.move(source, target, StandardCopyOption.REPLACE_EXISTING);
152 				}
153 				else
154 				{
155 					log.debug("Copying '" + sourceLocation + "' to '" + target + "'");
156 					Files.copy(source, target, StandardCopyOption.REPLACE_EXISTING);
157 				}
158 			}
159 			return true;
160 		}
161 	}
