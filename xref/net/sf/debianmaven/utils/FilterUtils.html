23  import org.apache.maven.model.Model;
24  import org.apache.maven.plugin.logging.Log;
25  
26  import java.io.IOException;
27  import java.nio.file.Files;
28  import java.nio.file.Path;
29  import java.nio.file.StandardCopyOption;
30  import java.nio.file.StandardOpenOption;
31  import java.util.ArrayList;
32  import java.util.HashMap;
33  import java.util.List;
34  import java.util.Map;

36  /**
37   * For filtering resources, replacing variables with actual values obtained from the model.
38   *
39   * @author FracPete (fracpete at waikato dot ac dot nz)
40   */
41  public class FilterUtils
42  {
43  
44  	public static final String PROJECT = "project.";
45  
46  	/**
47  	 * Extracts variables from a line: ${blah} -> blah.
48  	 *
49  	 * @param log for logging
50  	 * @param line the line to process
51  	 * @return the variables found
52  	 */
53  	protected static List<String> extractVariables(Log log, String line)
54  	{
55  		List<String> result;
56  		int indexStart;
57  		int indexEnd;
58  		String original;
59  
60  		original = line;
61  		result = new ArrayList<>();
62  		while (line.length() > 0)
63  		{
64  			indexStart = line.indexOf("${");
65  			if (indexStart == -1)
66  			{
67  				line = "";
68  			}
69  			else
70  			{
71  				if ((indexEnd = line.indexOf("}", indexStart)) == -1)
72  				{
73  					line = "";
74  				}
75  				else
76  				{
77  					result.add(line.substring(indexStart + 2, indexEnd));
78  					line = line.substring(indexEnd + 1);
79  				}
80  			}
81  		}
82  
83  		if (result.size() > 0)
84  			log.debug("Variables extracted from line '" + original + "': " + result);
85  
86  		return result;
87  	}
88  
89  	/**
90  	 * Obtains the variable value from the model ("project.*").
91  	 *
92  	 * @param log for logging
93  	 * @param var the variable
94  	 * @param model the model to obtain the value from
95  	 * @return null if not found or not a string object, otherwise the value
96  	 */
97  	protected static String variableValue(Log log, String var, Model model)
98  	{
99  		Object value;
100 
101 		if (!var.startsWith(PROJECT))
102 		{
103 			log.warn("Variable does not start with " + PROJECT + ": " + var);
104 			return null;
105 		}
106 
107 		value = BeanUtils.extractValue(log, model, var.substring(PROJECT.length()));
108 		if (value == null)
109 		{
110 			log.warn("Failed to extract value for variable: " + var);
111 			return null;
112 		}
113 		else
114 		{
115 			log.debug("Value for variable " + var + ": " + value);
116 			return "" + value;
117 		}
118 	}
119 
120 	/**
121 	 * Filters a file, replacing any project or additional variables in it.
122 	 *
123 	 * @param log for logging
124 	 * @param input the input file
125 	 * @param output the output file
126 	 * @param model the model to use for resolving "project.*" variables
127 	 * @param additional additional variables with associated values
128 	 * @throws IOException if reading/writing of file fails
129 	 */
130 	public static void filterFile(Log log, Path input, Path output, Model model, Map<String,String> additional) throws IOException
131 	{
132 		List<String> lines;
133 		int i;
134 		String line;
135 		List<String> vars;
136 		String value;
137 		boolean updated;
138 		boolean modified;
139 		Map<String,String> cache;
140 
141 		lines = Files.readAllLines(input);
142 		cache = new HashMap<>();
143 		modified = false;
144 		for (i = 0; i < lines.size(); i++)
145 		{
146 			line = lines.get(i);
147 			updated = false;
148 			vars = extractVariables(log, line);
149 			for (String var: vars)
150 			{
151 				if (cache.containsKey(var))
152 				{
153 					value = cache.get(var);
154 				}
155 				else if (additional.containsKey(var))
156 				{
157 					value = additional.get(var);
158 				}
159 				else
160 				{
161 					value = variableValue(log, var, model);
162 				}
163 				cache.put(var, value);
164 
165 				if (value != null)
166 				{
167 					log.debug("Replacing variable '" + var + "' with value '" + value + "' in: " + line);
168 					line = line.replace("${" + var + "}", value);
169 					updated = true;
170 				}
171 			}
172 			if (updated)
173 			{
174 				lines.set(i, line);
175 				modified = true;
176 			}
177 		}
178 
179 		if (modified)
180 		{
181 			log.debug("Writing filtered file " + input + " to " + output);
182 			Files.write(output, lines, StandardOpenOption.WRITE, StandardOpenOption.TRUNCATE_EXISTING, StandardOpenOption.CREATE);
183 		}
184 		else
185 		{
186 			log.debug("Copying unmodified file " + input + " to " + output);
187 			Files.copy(input, output, StandardCopyOption.REPLACE_EXISTING);
188 		}
189 	}
190 }
